This was a programming exercise in CSIS 223. It was used so that the program handles exceptions such as division by zero and invalid input. I really enjoyed C++ classes.

#include "stdafx.h" #include using namespace std; int main() { double x, y; double result; char op; cout << "Enter the 1st number, the operator, and the 2nd number seperated by spaces: " << endl; cin >> x >> op >> y; cout << endl; switch (op) { case '+': result = x + y; cout << "This is what you entered: " << x << " " << op << " " << y << " = " << result << endl; break; case '-': result = x - y; cout << "This is what you entered: " << x << " " << op << " " << y << " = " << result << endl; break; case '*': result = x * y; cout << "This is what you entered: " << x << " " << op << " " << y << " = " << result << endl; break; case '/': try { if (y == 0) throw 0; result = x / y; cout << x << " " << op << " " << y << " = " << result << endl; } catch (int) { cout << "Division by 0. " << endl; } break; default: cout << "You did not enter valid operation. " << endl; } system("pause"); return 0; }

This program that acts like a hand-held calculator or calculator app. It will include addition, subtraction, multiplication, division + results from a single calculation. Must include a menu to allow the user to choose what arithmetic operation they want to complete.

#include "stdafx.h" #include #include using namespace std; int main() { cout << fixed << showpoint << setprecision(10); char choice; double x, y; //symbol menu cout << "Enter the symbol you would like to use: [*] [/] [+] [-]: "; cin >> choice; //enter the numbers cout << " Enter the two numbers you would like to use Seperated by a space: "; cin >> x >> y; cout << endl; //switch for choice switch (choice) { case '*': cout << x * y << endl; break; case '/': cout << x / y << endl; break; case '+': cout << x + y << endl; break; case '-': cout << x - y << endl; break; default: //if the wrong symbol is entered cout << "You have chosen the wrong symbol "; cout << endl; return choice; break; } system("pause"); return 0; }
